PHASE 1.2

Quality Filter — Face & Video Validation

Loại bỏ video không phù hợp trước khi đưa vào solver — tiết kiệm GPU time

Đây là bước critical nhất về ROI: mỗi video không hợp lệ đưa vào UE5 solver sẽ lãng phí 3–10 phút GPU time. Filter pipeline chạy trên CPU và chiếm <2 giây/video — loại bỏ 40–60% corpus trước khi chạm đến solver.

Filter Tool Threshold Reject reason
Face detection RetinaFace ONNX ≥1 face, conf > 0.85 Không có face hoặc quá nhỏ (<15% frame)
Face size Bounding box area Face area > 10% frame Face quá xa camera — solver kém chất lượng
Multi-face Face count check Đúng 1 face chủ đạo >2 face làm solver không biết track ai
Blur/motion blur Laplacian variance Variance > 100 Video bị mờ — solver ra noise
Occlusion Facial landmark check <20% landmarks bị block Đeo khẩu trang, kính đen nặng, tay che mặt
Lighting Histogram analysis Mean brightness 40–220 Backlit, underexposed, overexposed
FPS ffprobe ≥24 fps Video slideshow hoặc low framerate
preprocessor/face_validator.py python
import cv2, numpy as np, onnxruntime as ort from pathlib import Path class FaceValidator: """ Chạy trên CPU — không chiếm VRAM của solver. Sample 10 frames đại diện, không đọc toàn bộ video. """ def __init__(self): # RetinaFace lightweight ONNX — 1.7MB, ~15ms/frame CPU self.detector = ort.InferenceSession( "models/retinaface_mobile.onnx", providers=["CPUExecutionProvider"] ) def validate_video(self, video_path: Path) -> dict: cap = cv2.VideoCapture(str(video_path)) total_frames = int(cap.get(cv2.CAP_PROP_FRAME_COUNT)) fps = cap.get(cv2.CAP_PROP_FPS) width = int(cap.get(cv2.CAP_PROP_FRAME_WIDTH)) height = int(cap.get(cv2.CAP_PROP_FRAME_HEIGHT)) result = { "valid": False, "fps": fps, "width": width, "height": height, "reject_reason": None } if fps < 24: result["reject_reason"] = f"low_fps:{fps:.1f}" return result # Sample 10 frames trải đều — tránh đọc toàn bộ video sample_positions = np.linspace(0.05, 0.95, 10) face_scores = [] blur_scores = [] for pos in sample_positions: cap.set(cv2.CAP_PROP_POS_FRAMES, int(pos * total_frames)) ret, frame = cap.read() if not ret: continue # Blur check — nhanh, chạy trước face detection gray = cv2.cvtColor(frame, cv2.COLOR_BGR2GRAY) blur_score = cv2.Laplacian(gray, cv2.CV_64F).var() blur_scores.append(blur_score) # Face detection faces = self._detect_faces(frame) face_scores.append(faces) cap.release() # Aggregate — dùng median để robust với outlier frames valid_face_frames = [f for f in face_scores if f["count"] == 1 and f["max_area_ratio"] > 0.08] face_ratio = len(valid_face_frames) / len(face_scores) if face_scores else 0 median_blur = np.median(blur_scores) if blur_scores else 0 if face_ratio < 0.6: result["reject_reason"] = f"face_coverage:{face_ratio:.2f}" elif median_blur < 80: result["reject_reason"] = f"blur:{median_blur:.1f}" else: result["valid"] = True return result def _detect_faces(self, frame: np.ndarray) -> dict: # Resize để tăng tốc detection, giữ aspect ratio h, w = frame.shape[:2] scale = 640 / max(h, w) small = cv2.resize(frame, (int(w*scale), int(h*scale))) # ... run ONNX inference ... # Trả về dict với face boxes, confidence, area ratio return {"count": 1, "max_area_ratio": 0.25, "conf": 0.95}

import imagehash from PIL import Image import cv2, numpy as np, sqlite3 class VideoDeduplicator: """ 2 cấp dedup: 1. Video-level: perceptual hash của keyframes — bắt repost, duplicate upload 2. Identity-level: face embedding clustering — giới hạn N video/người """ def video_phash(self, video_path) -> str: # Lấy frame ở 10%, 50%, 90% — hash từng frame, concat hashes = [] cap = cv2.VideoCapture(str(video_path)) total = int(cap.get(cv2.CAP_PROP_FRAME_COUNT)) for pos in [0.1, 0.5, 0.9]: cap.set(cv2.CAP_PROP_POS_FRAMES, int(pos * total)) ret, frame = cap.read() if ret: img = Image.fromarray(cv2.cvtColor(frame, cv2.COLOR_BGR2RGB)) hashes.append(str(imagehash.phash(img, hash_size=16))) cap.release() return "_".join(hashes) def is_duplicate(self, phash: str, db, threshold=8) -> bool: # Hamming distance < threshold → duplicate existing = db.execute("SELECT phash FROM videos WHERE status != 'rejected'") h1 = imagehash.hex_to_hash(phash.split("_")[1]) for (h_str,) in existing: if h_str: h2 = imagehash.hex_to_hash(h_str.split("_")[1]) if h1 - h2 < threshold: return True return False def face_embedding(self, video_path, face_model) -> np.ndarray: # ArcFace embedding — 512-dim vector đại diện identity # Dùng để cluster — giới hạn 100 video/person max cap = cv2.VideoCapture(str(video_path)) total = int(cap.get(cv2.CAP_PROP_FRAME_COUNT)) cap.set(cv2.CAP_PROP_POS_FRAMES, int(0.5 * total)) ret, frame = cap.read() cap.release() if not ret: return None return face_model.get_embedding(frame) # InsightFace
PHASE 2.1

Video Preprocessing — Segment & Normalize

Chuẩn hoá video trước khi đưa vào MHA solver — critical cho chất lượng output

⚠
Giới hạn quan trọng của MHA Mono Video Mode
MetaHuman Animator (mono mode) hoạt động tốt nhất với video: H264/H265 MP4, 1080p hoặc 720p, 30fps cố định, không bị scene cut trong đoạn. Video dài (>5 phút) cần được segment thành đoạn ≤3 phút để tránh solver drift và memory leak của UE5.
preprocessor/video_normalize.py python
import subprocess, json from pathlib import Path class VideoNormalizer: """ Chuẩn hoá video cho MHA solver: - Segment tại scene cut — tránh solver nhầm identity - Cố định 30fps — solver expect constant framerate - Scale về 1080p max — không upscale, chỉ downscale - Audio: extract WAV 48kHz mono riêng cho audio solver """ SEGMENT_MAX_SEC = 180 # 3 phút max per segment TARGET_FPS = 30 TARGET_HEIGHT = 1080 def detect_scene_cuts(self, video_path: Path) -> list[float]: """PySceneDetect — tìm hard cuts.""" result = subprocess.run([ "scenedetect", "-i", str(video_path), "detect-content", "--threshold", "27", "list-scenes", "-o", "/dev/null" ], capture_output=True, text=True) # Parse output để lấy timestamps # ... returns list of cut timestamps in seconds return [] def segment_and_normalize( self, video_path: Path, out_dir: Path ) -> list[Path]: """ Segment video tại scene cuts, normalize mỗi segment. Return list of output segment paths. """ cuts = self.detect_scene_cuts(video_path) # Chia thêm nếu segment dài hơn SEGMENT_MAX_SEC segments = self._make_time_windows( video_path, cuts, self.SEGMENT_MAX_SEC ) out_paths = [] for i, (start, end) in enumerate(segments): out_path = out_dir / f"seg_{i:04d}.mp4" self._ffmpeg_normalize(video_path, out_path, start, end) out_paths.append(out_path) return out_paths def _ffmpeg_normalize( self, inp: Path, out: Path, start: float, end: float ): duration = end - start vf_filters = [ # Scale: giữ aspect ratio, chỉ downscale f"scale='min({self.TARGET_HEIGHT},ih)':'min({self.TARGET_HEIGHT},iw)'" ":force_original_aspect_ratio=decrease", # Pad thành multiple of 2 (H264 requirement) "pad=ceil(iw/2)*2:ceil(ih/2)*2", # Cố định fps — MHA cần constant framerate f"fps={self.TARGET_FPS}", ] subprocess.run([ "ffmpeg", "-y", "-ss", str(start), "-t", str(duration), "-i", str(inp), "-vf", ",".join(vf_filters), "-c:v", "libx264", "-crf", "18", # chất lượng cao — solver nhạy với artifacts "-preset", "fast", "-pix_fmt", "yuv420p", "-an", # audio extract riêng ở phase 2.2 str(out) ], check=True)

# ═══════════════════════════════════════════════════════════════ # MetaHuman Animator — Production Batch Solver # Chạy trong UE5 headless, xử lý queue liên tục 24/7 # ═══════════════════════════════════════════════════════════════ import unreal, sqlite3, time, json, logging, traceback from pathlib import Path # ─── Config ──────────────────────────────────────────────────── DB_PATH = r"D:\pipeline\jobs.db" VIDEO_STAGE_DIR = r"D:\pipeline\stage" # normalized segments OUTPUT_UASSET = r"D:\pipeline\ue5_output" # AnimSequence assets UE_CONTENT_ROOT = "/Game/BatchOutput" LOG_FILE = r"D:\pipeline\logs\solver.log" POLL_INTERVAL = 5 # giây — poll DB khi không có job GC_EVERY = 50 # garbage collect sau N jobs RESTART_EVERY = 500 # restart UE5 process sau N jobs (memory safety) logging.basicConfig( filename=LOG_FILE, level=logging.INFO, format='%(asctime)s [%(levelname)s] %(message)s' ) class BatchSolverSession: def __init__(self): self.db = sqlite3.connect(DB_PATH, check_same_thread=False) self.jobs_processed = 0 self.asset_tools = unreal.AssetToolsHelpers.get_asset_tools() def run_forever(self): """Main loop — chạy cho đến khi RESTART_EVERY jobs.""" unreal.log("[Batch] Solver session started.") while self.jobs_processed < RESTART_EVERY: jobs=self. _get_pending_jobs(batch_size=1) if not jobs: unreal.log("[Batch] Queue empty. Polling...") time.sleep(POLL_INTERVAL) continue for job in jobs: self._process_job(job) self.jobs_processed += 1 if self.jobs_processed % GC_EVERY == 0: self._force_gc() unreal.log(f"[Batch] {RESTART_EVERY} jobs done. Requesting restart.") # Ghi flag để watchdog biết cần restart (không phải crash) Path(r"D:\pipeline\flags\restart_requested").touch() def _process_job(self, job: dict): vid_id = job["id"] seg_path = Path(job["segment_path"]) wav_path = Path(job["wav_path"]) unreal.log(f"[Batch] Processing: {vid_id}") self._update_status(vid_id, "solving") try: # ── Step 1: Ingest video → CaptureData asset ────────── capture_data = self._ingest_capture_data( vid_id, seg_path, wav_path ) # ── Step 2: Create MetaHumanPerformance asset ───────── perf_path = f"{UE_CONTENT_ROOT}/Perf_{vid_id}" perf = self.asset_tools.create_asset( f"Perf_{vid_id}", UE_CONTENT_ROOT, unreal.MetaHumanPerformance, unreal.MetaHumanPerformanceFactoryNew() ) # Config: mono mode, không cần depth data perf.set_editor_property("capture_data", capture_data) perf.set_editor_property("input_type", unreal.MetaHumanPerformanceInputType.MONOCULAR_FOOTAGE) # Audio solver — enable để lấy tongue/jaw curves từ audio perf.set_editor_property("use_audio", True) # ── Step 3: Run solve ────────────────────────────────── t_start = time.time() perf.process_performance() solve_time = time.time() - t_start unreal.log(f"[Batch] Solved in {solve_time:.1f}s: {vid_id}") # ── Step 4: Export AnimSequence ─────────────────────── anim_path = self._export_animation(perf, vid_id) # ── Step 5: Cleanup interim assets ──────────────────── self._cleanup_assets(capture_data, perf) self._update_status(vid_id, "solved", anim_uasset=anim_path, solve_time=solve_time) except Exception as e: err = traceback.format_exc() logging.error(f"Job failed {vid_id}: {err}") self._update_status(vid_id, "failed", error=str(e)[:512]) def _ingest_capture_data( self, vid_id: str, seg_path: Path, wav_path: Path ) -> unreal.Object: """ Tạo MonoCaptureData asset từ video file. Đây là bước UE5 copy video vào project — cần đủ disk space. """ factory = unreal.MetaHumanMonoCaptureDataFactory() factory.set_editor_property("source_footage_path", str(seg_path)) factory.set_editor_property("source_audio_path", str(wav_path)) capture = self.asset_tools.create_asset( f"Cap_{vid_id}", UE_CONTENT_ROOT, unreal.MetaHumanMonoCaptureData, factory ) return capture def _export_animation(self, perf, vid_id: str) -> str: """Export AnimSequence — chứa 135+ facial curves.""" anim_name = f"Anim_{vid_id}" perf.set_editor_property( "output_animation_asset_path", f"{UE_CONTENT_ROOT}/{anim_name}" ) perf.export_animation( unreal.PerformanceExportRange.WHOLE_SEQUENCE ) # Save asset to disk unreal.EditorAssetLibrary.save_asset( f"{UE_CONTENT_ROOT}/{anim_name}" ) return f"{UE_CONTENT_ROOT}/{anim_name}" def _cleanup_assets(self, *assets): """Xoá interim assets để tránh disk và memory bloat.""" for asset in assets: path = asset.get_path_name() unreal.EditorAssetLibrary.delete_asset(path) def _force_gc(self): """Force UE5 garbage collection — chạy mỗi N jobs.""" unreal.SystemLibrary.execute_console_command( None, "gc.CollectGarbageEveryFrame 1" ) time.sleep(2) unreal.SystemLibrary.execute_console_command( None, "gc.CollectGarbageEveryFrame 0" ) def _get_pending_jobs(self, batch_size=1) -> list: cur = self.db.execute( """SELECT id, segment_path, wav_path FROM segments WHERE status = 'preprocessed' ORDER BY priority DESC, created_at ASC LIMIT ?""", (batch_size,) ) return [dict(row) for row in cur.fetchall()] def _update_status(self, vid_id: str, status: str, **kwargs): fields = ", ".join([f"{k}=?" for k in kwargs]) vals = list(kwargs.values()) + [status, vid_id] sql = f"UPDATE segments SET {fields}, status=? WHERE id=?" \ if fields else "UPDATE segments SET status=? WHERE id=?" self.db.execute(sql, vals) self.db.commit() # Entry point — được gọi bởi UE5 -run=PythonScript session = BatchSolverSession() session.run_forever()

postprocessor/validate_output.py python
import numpy as np from pathlib import Path from scipy.signal import butter, filtfilt class CurveValidator: """ Sau khi extract curves, detect các failure modes của MHA solver: 1. Solver drift — curves đột ngột nhảy về 0 hoặc về max 2. Frozen curves — không thay đổi trong đoạn dài (solver stuck) 3. Jitter — noise tần số cao không tự nhiên (solver uncertain) 4. Range violation — giá trị ngoài [0, 1] bình thường """ JITTER_THRESHOLD = 0.15 # std dev per frame — quá cao = jitter FROZEN_THRESHOLD = 60 # số frames lien tiếp không đổi DRIFT_THRESHOLD = 0.8 # jump > 0.8 in 1 frame = drift def validate(self, npz_path: Path) -> dict: data = np.load(npz_path, allow_pickle=False) results = {"valid": True, "issues": [], "quality_score": 1.0} curve_names = [k for k in data.files if k.startswith("CTRL_")] for name in curve_names: curve = data[name] # Check jitter diff = np.abs(np.diff(curve)) if np.mean(diff) > self.JITTER_THRESHOLD: results["issues"].append( {"type": "jitter", "curve": name, "severity": float(np.mean(diff))} ) results["quality_score"] -= 0.02 # Check frozen segments frozen_runs = self._count_frozen_runs(curve) if frozen_runs > self.FROZEN_THRESHOLD: results["issues"].append( {"type": "frozen", "curve": name, "max_run": frozen_runs} ) # Check drift max_jump = np.max(diff) if len(diff) > 0 else 0 if max_jump > self.DRIFT_THRESHOLD: results["issues"].append( {"type": "drift", "curve": name, "max_jump": float(max_jump)} ) results["quality_score"] -= 0.05 results["quality_score"] = max(0, results["quality_score"]) results["valid"] = results["quality_score"] > 0.5 return results def smooth_curves(self, npz_path: Path, out_path: Path, cutoff_hz=15): """ Low-pass filter — loại bỏ jitter tần số cao. Giữ lại chuyển động tự nhiên (0.5–12 Hz). Cutoff 15 Hz là conservative — giữ micro-expressions. """ data = np.load(npz_path, allow_pickle=False) fps = float(data["fps"][0]) nyq = fps / 2 b, a = butter(4, cutoff_hz / nyq, btype="low") smoothed = {} for key in data.files: if key.startswith("CTRL_") and data[key].ndim == 1: smoothed[key] = filtfilt(b, a, data[key]) else: smoothed[key] = data[key] np.savez_compressed(out_path, **smoothed)
